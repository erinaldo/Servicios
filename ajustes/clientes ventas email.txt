select sum(if(tblventas.idconversion=2,round(tblventas.totalapagar,2),round(tblventas.totalapagar*tblventas.tipodecambio,2))) as totalapagar,tblclientes.nombre as cnombre,tblclientes.email from tblventas inner join tblclientes on tblventas.idcliente=tblclientes.idcliente where tblventas.fecha>='2016/01/01' and tblventas.fecha<='2016/12/30' and tblventas.estado=3 group by tblclientes.idcliente
