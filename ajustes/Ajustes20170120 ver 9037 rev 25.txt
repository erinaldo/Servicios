CREATE DEFINER=`root`@`localhost` FUNCTION `spmodificainventarioubicacionesf`(pidinventario int,pidalmacen int,pcantidad double,pcantidadant double,pmodo tinyint,pmedida tinyint, pubicacion varchar(15)) RETURNS int(11)
BEGIN

declare vcontenido double;
declare vcantidad double;
declare vesinventariable tinyint;
if (select inventariable from tblinventario where idinventario=pidinventario)=1 then
  set vcantidad=ifnull((select ifnull(u.cantidad,0) from tblalmacenesi a left outer join tblalmacenesiubicaciones u on a.id=u.idalmaceni where idinventario=pidinventario and idalmacen=pidalmacen and ubicacion=pubicacion),0);
  if pmedida=0 then
    set vcontenido=(select contenido from tblinventario where idinventario=pidinventario);
    if vcontenido<>0 then
      if pmodo=0 then
        set vcantidad=vcantidad+pcantidad/vcontenido;
      end if;
      if pmodo=1 then
        set vcantidad=vcantidad-pcantidad/vcontenido;
      end if;
      if pmodo=2 then
        set vcantidad=vcantidad-pcantidadant/vcontenido+pcantidad/vcontenido;
      end if;
      if pmodo=3 then
        set vcantidad=vcantidad+pcantidadant/vcontenido-pcantidad/vcontenido;
      end if;
    end if;
  else
    if pmodo=0 then
      set vcantidad=vcantidad+pcantidad;
    end if;
    if pmodo=1 then
      set vcantidad=vcantidad-pcantidad;
    end if;
    if pmodo=2 then
      set vcantidad=vcantidad-pcantidadant+pcantidad;
    end if;
    if pmodo=3 then
      set vcantidad=vcantidad+pcantidadant-pcantidad;
    end if;
  end if;
  if (select count(*) from tblalmacenesiubicaciones u inner join tblalmacenesi a on u.idalmaceni=a.id where a.idinventario=pidinventario and a.idalmacen=pidalmacen and u.ubicacion=pubicacion)=0 then
    insert into tblalmacenesiubicaciones (idalmaceni, cantidad, ubicacion) select id, vcantidad, pubicacion from tblalmacenesi where idinventario=pidinventario and idalmacen=pidalmacen;
  else
    update tblalmacenesiubicaciones inner join tblalmacenesi on tblalmacenesiubicaciones.idalmaceni=tblalmacenesi.id set tblalmacenesiubicaciones.cantidad=vcantidad where tblalmacenesi.idinventario=pidinventario and idalmacen=pidalmacen and ubicacion=pubicacion;
  end if;
end if;
return 1;

END $$
